**Step 1: Create ScrollReveal component (SCRL-01)**

Create `src/components/animations/ScrollReveal.tsx` as a NEW client component (do NOT modify FadeIn.tsx -- it's used throughout the app and its API should remain stable):

```tsx
"use client";

import { motion } from "motion/react";

type Direction = "up" | "left" | "right";

const offsets: Record<Direction, { x: number; y: number }> = {
  up: { x: 0, y: 24 },
  left: { x: -24, y: 0 },
  right: { x: 24, y: 0 },
};

interface ScrollRevealProps {
  children: React.ReactNode;
  direction?: Direction;
  delay?: number;
  className?: string;
}

export function ScrollReveal({ children, direction = "up", delay = 0, className }: ScrollRevealProps) {
  const offset = offsets[direction];
  return (
    <motion.div
      initial={{ opacity: 0, x: offset.x, y: offset.y }}
      whileInView={{ opacity: 1, x: 0, y: 0 }}
      viewport={{ once: true, amount: 0.3 }}
      transition={{ duration: 0.5, delay, ease: "easeOut" }}
      className={className}
    >
      {children}
    </motion.div>
  );
}
```

This is intentionally similar to FadeIn but adds the `direction` prop for directional variety. Creating a separate component avoids breaking existing FadeIn usage throughout the app.

**Step 2: Create ScrollProgress component (SCRL-02)**

Create `src/components/animations/ScrollProgress.tsx` as a client component:

```tsx
"use client";

import { motion, useScroll, useSpring } from "motion/react";

export function ScrollProgress() {
  const { scrollYProgress } = useScroll();
  const scaleX = useSpring(scrollYProgress, {
    stiffness: 100,
    damping: 30,
    restDelta: 0.001,
  });

  return (
    <motion.div
      className="fixed top-0 left-0 right-0 h-1 bg-primary origin-left z-[60]"
      style={{ scaleX }}
    />
  );
}
```

The `z-[60]` ensures it renders above the TopNav (z-50). `origin-left` ensures the bar grows from left to right. `useSpring` adds smooth interpolation to the scroll progress value.

**Step 3: Create ParallaxHero component (SCRL-03)**

Create `src/components/animations/ParallaxHero.tsx` as a client component:

```tsx
"use client";

import { motion, useScroll, useTransform } from "motion/react";
import { useRef } from "react";

interface ParallaxHeroProps {
  children: React.ReactNode;
  className?: string;
}

export function ParallaxHero({ children, className }: ParallaxHeroProps) {
  const ref = useRef(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ["start start", "end start"],
  });
  const y = useTransform(scrollYProgress, [0, 1], [0, 60]);

  return (
    <div ref={ref} className={className}>
      {/* Hide on lg+ where sticky layout conflicts with parallax (Pitfall 5) */}
      {/* On mobile: parallax effect. On desktop: static (no transform). */}
      <motion.div className="lg:!transform-none" style={{ y }}>
        {children}
      </motion.div>
    </div>
  );
}
```

Key design decision per Pitfall 5 from RESEARCH.md: The detail page left column uses `lg:sticky lg:top-20`. Parallax inside a sticky container causes jarring scroll behavior. Solution: Use `lg:!transform-none` to disable the transform on desktop, keeping parallax mobile-only. The `!` makes it important to override the inline style from Motion.

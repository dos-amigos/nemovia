**In supabase/functions/scrape-sagre/index.ts:**

1. Add an `isNoiseTitle(title: string): boolean` function BEFORE the `extractRawEvent` function. This function detects non-event entries that scrapers pick up. It should return `true` (noise) if ANY of these conditions match:

   - Title length < 5 characters (too short to be a real event)
   - Title length > 150 characters (likely a page description or navigation text)
   - Title matches known noise patterns (case-insensitive):
     - Contains "calendario" AND ("mensile" OR "regioni" OR "italiane")
     - Contains "cookie" or "privacy policy" or "termini"
     - Contains "cerca sagr" or "ricerca event"
     - Starts with "menu" or "navigazione" or "home"
     - Is entirely numeric or entirely punctuation
     - Contains "tutte le sagre" or "elenco sagre" or "lista sagre"
   - Date span suggests non-event: if the title contains a date range spanning > 180 days (like "1 Gennaio - 31 Dicembre"), it's a calendar page. However, this check is complex — instead, add a simpler heuristic: if title contains both "gennaio" and "dicembre" (case-insensitive), flag as noise.

   ```typescript
   function isNoiseTitle(title: string): boolean {
     if (!title || title.length < 5 || title.length > 150) return true;
     const t = title.toLowerCase();
     // Known noise patterns
     if (/calendario\s.*(mensile|regioni|italian)/i.test(t)) return true;
     if (/cookie|privacy\s*policy|termini\s*(e\s*)?condizion/i.test(t)) return true;
     if (/cerca\s+sagr|ricerca\s+event/i.test(t)) return true;
     if (/^(menu|navigazione|home)\b/i.test(t)) return true;
     if (/^[\d\s\-\/\.]+$/.test(title.trim())) return true; // all numeric/punct
     if (/tutte le sagre|elenco sagre|lista sagre/i.test(t)) return true;
     if (/gennaio.*dicembre|dicembre.*gennaio/i.test(t)) return true;
     return false;
   }
   ```

   ```typescript
   /**
    * Normalize location_text for better Nominatim geocoding results.
    * Handles: province codes, region prefixes, extra whitespace, common noise.
    */
   function normalizeLocationText(raw: string): string {
     let s = raw.trim();
     if (!s) return "";
     // Remove province codes: "(VR)", "- VR", ", VR"
     s = s.replace(/\s*\([A-Z]{2}\)\s*/g, "");
     s = s.replace(/\s*[-,]\s*[A-Z]{2}\s*$/g, "");
     // Remove region prefixes: "Veneto ", "Veneto - ", etc.
     s = s.replace(/^(Veneto|Lombardia|Piemonte|Emilia[\s-]?Romagna|Trentino|Friuli[\s-]?Venezia[\s-]?Giulia)\s*[-:,]?\s*/i, "");
     // Remove "Provincia di " prefix
     s = s.replace(/^Provincia\s+di\s+/i, "");
     // Collapse multiple spaces
     s = s.replace(/\s+/g, " ").trim();
     // Append ", Veneto, Italy" for Nominatim disambiguation if the result is a bare city name
     // (helps Nominatim resolve ambiguous Italian city names to Veneto specifically)
     if (s && !s.includes(",")) {
       s = s + ", Veneto";
     }
     return s;
   }

   // Keep cleanCityName as alias for compatibility
   function cleanCityName(raw: string): string {
     return normalizeLocationText(raw);
   }
   ```

```typescript
function isNoiseTitle(title: string): boolean {
  if (!title || title.length < 5 || title.length > 150) return true;
  const t = title.toLowerCase();
  if (/calendario\s.*(mensile|regioni|italian)/i.test(t)) return true;
  if (/cookie|privacy\s*policy|termini\s*(e\s*)?condizion/i.test(t)) return true;
  if (/cerca\s+sagr|ricerca\s+event/i.test(t)) return true;
  if (/^(menu|navigazione|home)\b/i.test(t)) return true;
  if (/^[\d\s\-\/\.]+$/.test(title.trim())) return true;
  if (/tutte le sagre|elenco sagre|lista sagre/i.test(t)) return true;
  if (/gennaio.*dicembre|dicembre.*gennaio/i.test(t)) return true;
  return false;
}
```

  <behavior>
    isNoiseTitle(title: string) -> boolean:
    - Returns true for empty/null/too-short (<5 chars)/too-long (>150 chars) titles
    - Returns true for calendar spam: "Calendario mensile eventi sagre Gennaio 2026", "Calendario eventi e sagre in Veneto"
    - Returns true for navigation noise: "Cookie Policy", "Menu", "Cerca sagre"
    - Returns true for new spam patterns: "Programma completo", "Scopri tutte le...", "Newsletter", "Iscriviti"
    - Returns true for aggregator noise: "Tutte le sagre", "Elenco sagre"
    - Returns true for numeric-only titles: "01/02/2026"
    - Returns false for legitimate titles: "Sagra del Baccala alla Vicentina", "Festa della Polenta", "Sagra del Pesce di Chioggia"
    - Edge case: "Sagra della Polenta - Calendario 2026" should NOT be rejected (real sagra with "Calendario" in a non-spam context)

    isCalendarDateRange(startDate: string | null, endDate: string | null) -> boolean:
    - Returns true when start is day 1 AND end is day >= 28 of same or any month (full-month calendar spam)
    - Returns true for "2026-01-01" to "2026-01-31" (full January)
    - Returns true for "2026-02-01" to "2026-02-28" (full February)
    - Returns false for "2026-04-24" to "2026-04-26" (normal 3-day sagra)
    - Returns false for "2026-06-01" to "2026-06-03" (starts on 1st but ends on 3rd, not 28+)
    - Returns false when either date is null

    isExcessiveDuration(startDate: string | null, endDate: string | null, maxDays?: number) -> boolean:
    - Default maxDays = 7
    - Returns true when duration exceeds maxDays (e.g., "2026-01-01" to "2026-01-15" = 14 days)
    - Returns false for duration <= maxDays (e.g., "2026-04-24" to "2026-04-26" = 2 days)
    - Returns false for exactly maxDays (e.g., 7-day span is allowed)
    - Returns false when either date is null

    isPastYearEvent(startDate: string | null, endDate: string | null) -> boolean:
    - Returns true when start_date year < current year (2025 events in 2026)
    - Returns true when end_date year < current year
    - Returns false for current year events
    - Returns false when both dates are null
  </behavior>

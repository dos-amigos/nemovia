2. **Create `src/components/animations/FadeIn.tsx`**: A "use client" wrapper that animates children on scroll into view:
   ```typescript
   "use client";
   import { motion } from "motion/react";

   interface FadeInProps {
     children: React.ReactNode;
     delay?: number;
     className?: string;
   }

   export function FadeIn({ children, delay = 0, className }: FadeInProps) {
     return (
       <motion.div
         initial={{ opacity: 0, y: 24 }}
         whileInView={{ opacity: 1, y: 0 }}
         viewport={{ once: true, amount: 0.3 }}
         transition={{ duration: 0.5, delay, ease: "easeOut" }}
         className={className}
       >
         {children}
       </motion.div>
     );
   }
   ```
   - `viewport.once: true` prevents re-animation on subsequent scrolls.
   - `amount: 0.3` triggers when 30% of element is visible.
   - Accepts optional `delay` for staggering sibling sections.
   - Accepts optional `className` for pass-through styling.

3. **Create `src/components/animations/StaggerGrid.tsx`**: A "use client" wrapper that reveals children with staggered spring animation:
   ```typescript
   "use client";
   import React from "react";
   import { motion } from "motion/react";

   const container = {
     hidden: {},
     show: { transition: { staggerChildren: 0.08 } },
   };

   const item = {
     hidden: { opacity: 0, y: 20 },
     show: { opacity: 1, y: 0, transition: { type: "spring", damping: 20, stiffness: 200 } },
   };

   interface StaggerGridProps {
     children: React.ReactNode;
     className?: string;
   }

   export function StaggerGrid({ children, className = "grid grid-cols-1 gap-4 sm:grid-cols-2" }: StaggerGridProps) {
     return (
       <motion.div
         variants={container}
         initial="hidden"
         whileInView="show"
         viewport={{ once: true, amount: 0.1 }}
         className={className}
       >
         {React.Children.map(children, (child) => (
           <motion.div variants={item}>{child}</motion.div>
         ))}
       </motion.div>
     );
   }
   ```
   - Default className matches existing SagraGrid layout.
   - Spring physics (damping 20, stiffness 200) for natural motion.
   - `staggerChildren: 0.08` for fast but visible sequential reveal.
   - `amount: 0.1` triggers early since grids are often large.
  </action>
